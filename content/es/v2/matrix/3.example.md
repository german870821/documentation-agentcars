Angularjson
---------------------
```json
  var host_base_url = "https://api.dev.agentcars.com/v2/";

  $http.get(host_base_url + 'sites/get-matrix', {
        params: {
            "id": 100,
            "locationName": $routeParams.locationName,
            "locationName2": $routeParams.locationName2,
            "pickup": $routeParams.pickup,
            "dropoff": $routeParams.dropoff,
            "phour": $routeParams.phour,
            "dhour": $routeParams.dhour,
            "include": $routeParams.include,
            "office": $routeParams.office,
            "office2": $routeParams.office2,
            "lat": $routeParams.lat,
            "long": $routeParams.long,
            "con": $routeParams.con,
            "emp": $routeParams.emp,
            "codigo_pc": $routeParams.codigo_pc,
            "codigo_cd": $routeParams.codigo_cd,
        }
    }).success(function (data) {
        $scope.matrixInfo = data;
    }).error(function () {
        console.debug("error");
    });

```
